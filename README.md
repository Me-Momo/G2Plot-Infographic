## Install

```bash
$ npm i --save d2-infographic
```


## Usage

 - render

```ts
import { G2Plot } from '@antv/g2plot';
import { adaptor, defaultOptions } from 'd2-infographic';

const infographic = new G2Plot('container', {
  centralText: {
    content: 'Hi, AntV',
    style: {
      fontSize: 24
    }
  }
}, adaptor, defaultOptions);

infographic.render();
```

 - update

```ts
infographic.update({
  music: { type: 'metal' },
});
```
