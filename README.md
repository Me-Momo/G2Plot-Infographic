## Install

```bash
$ npm i --save g2plot-infographic
```


## Usage

 - render

```ts
import { G2Plot } from '@antv/g2plot';
import { adaptor, defaultOptions } from 'g2plot-infographic';

const infographic = new G2Plot('container', {
  centralText: {
    content: 'Hi, AntV',
    style: {
      fontSize: 24
    }
  }
}, adaptor, defaultOptions);

infographic.render();
```

 - update

```ts
infographic.update({
  music: { type: 'metal' },
});
```
